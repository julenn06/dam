-- Crear la base de datos
CREATE DATABASE empresa;

-- Usar la base de datos
USE empresa;

-- Crear tabla DEPARTAMENTOS
CREATE TABLE DEPARTAMENTOS (
    DEPT_NO INT PRIMARY KEY,
    DNOMBRE VARCHAR(50),
    LOC VARCHAR(50)
);

-- Crear tabla EMPLEADOS
CREATE TABLE EMPLEADOS (
    EMP_NO INT PRIMARY KEY,
    APELLIDO VARCHAR(50),
    OFICIO VARCHAR(50),
    DIR INT,
    FECHA_ALT DATE,
    SALARIO FLOAT,
    COMISION FLOAT,
    DEPT_NO INT,
    FOREIGN KEY (DIR) REFERENCES EMPLEADOS(EMP_NO),
    FOREIGN KEY (DEPT_NO) REFERENCES DEPARTAMENTOS(DEPT_NO)
);

-- Procedimientos almacenados para consultas
DELIMITER //
CREATE PROCEDURE consultar_departamentos()
BEGIN
    SELECT * FROM DEPARTAMENTOS;
END //
DELIMITER ;

DELIMITER //
CREATE PROCEDURE consultar_empleados_dept(IN dept INT)
BEGIN
    SELECT APELLIDO, OFICIO, SALARIO FROM EMPLEADOS WHERE DEPT_NO = dept;
END //
DELIMITER ;

DELIMITER //
CREATE PROCEDURE consultar_empleado_max_salario()
BEGIN
    SELECT APELLIDO, OFICIO, SALARIO FROM EMPLEADOS WHERE SALARIO = (SELECT MAX(SALARIO) FROM EMPLEADOS);
END //
DELIMITER ;

-- Procedimiento almacenado para añadir departamento MARKETING y empleados
DELIMITER //
CREATE PROCEDURE añadir_dept_marketing()
BEGIN
    IF NOT EXISTS (SELECT 1 FROM DEPARTAMENTOS WHERE DEPT_NO = 50) THEN
        INSERT INTO DEPARTAMENTOS VALUES (50, 'MARKETING', 'VALENCIA');
    END IF;
    IF NOT EXISTS (SELECT 1 FROM EMPLEADOS WHERE EMP_NO = 8000) THEN
        INSERT INTO EMPLEADOS VALUES (8000, 'PEREZ', 'ANALISTA', NULL, '2024-01-01', 3200, 0, 50);
    END IF;
    IF NOT EXISTS (SELECT 1 FROM EMPLEADOS WHERE EMP_NO = 8001) THEN
        INSERT INTO EMPLEADOS VALUES (8001, 'LOPEZ', 'EMPLEADO', 8000, '2024-01-15', 1500, 200, 50);
    END IF;
END //
DELIMITER ;

-- Procedimiento almacenado para actualizar salarios en MARKETING
DELIMITER //
CREATE PROCEDURE actualizar_salarios_marketing(IN salario_nuevo FLOAT)
BEGIN
    UPDATE EMPLEADOS SET SALARIO = salario_nuevo WHERE DEPT_NO = 50;
END //
DELIMITER ;

-- Procedimiento almacenado para eliminar departamento MARKETING y empleados
DELIMITER //
CREATE PROCEDURE eliminar_dept_marketing()
BEGIN
    -- Primero eliminar empleados que no son jefes (DIR no referenciado)
    DELETE FROM EMPLEADOS WHERE DEPT_NO = 50 AND EMP_NO NOT IN (SELECT DISTINCT DIR FROM EMPLEADOS WHERE DIR IS NOT NULL);
    -- Luego eliminar los jefes
    DELETE FROM EMPLEADOS WHERE DEPT_NO = 50;
    -- Finalmente eliminar el departamento
    DELETE FROM DEPARTAMENTOS WHERE DEPT_NO = 50;
END //
DELIMITER ;
